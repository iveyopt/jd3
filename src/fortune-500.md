    Plot.dot(data, {

      y: "redditorBrandCakeday",

      dx: -2,

    Plot.crosshair(data, {
      x: "rank",
      y: "founded"

    Plot.tip(data, Plot.pointerX({

      title: (d) => `${d.company} \n Founded: ${d.founded} \n Joined Reddit: ${d.redditorBrandCakeday}`
